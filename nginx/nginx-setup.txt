upstream api {
    server backend:8000;
}

server {
  listen 8080;

  location / {
    root /var/www/frontend;
    index index.html;
    try_files $uri /index.html$is_args$args =404;
  }

  location /api/ {
    proxy_pass http://api;
    proxy_set_header Host $http_host;
  }

  location /static/ {
        alias /home/app/backend/staticfiles/;
    }


}
